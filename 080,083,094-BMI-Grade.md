def calculate_gpa(self):
    grade_to_gpa = {'A': 4.0, 'B+': 3.5, 'B': 3.0, 'C+': 2.5, 
                    'C': 2.0, 'D+': 1.5, 'D': 1.0, 'F': 0.0}
    
    total_weight = 0
    total_score = 0
    
    for i in range(len(self.subject)):
        grade = self.calculate_grade(int(self.subject[i].get()))
        gpa = grade_to_gpa[grade]
        weight = float(self.weight[i].get())

        total_score += gpa * weight
        total_weight += weight

    gpa = total_score / total_weight
    print(f"GPA ‡∏Ç‡∏≠‡∏á‡∏Ñ‡∏∏‡∏ì‡∏Ñ‡∏∑‡∏≠: {gpa:.2f}")
